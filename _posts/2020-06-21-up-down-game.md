## python code

{% highlight css %}
import random
count=0
guess=0
number=random.randint(1,100)
print("< UP&DOWN GAME >")

while guess != number:
    count=count+1
    guess=int(input("Please enter a number : "))
    if guess>number:
        print("DOWN")
    elif guess<number:
        print("UP")
    else :
        print("CONGRATULATIONS! COUNT = ",count)
        break
{% endhighlight %}
